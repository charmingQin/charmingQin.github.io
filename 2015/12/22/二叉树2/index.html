先序遍历:

// 首先访问根，再先序遍历左（右）子树，最后先序遍历右（左）子树
void preOrderTree(BinaryTree *tree)
{
if(tree == NULL) return;
else
{
    printf("%3d",tree -> root);
    preOrderTree(tree -> leftTree);
    preOrderTree(tree -> rightTree);
}
}

中序遍历:

//首先中序遍历左（右）子树，再访问根，最后中序遍历右（左）子树
void middleOrderTree(BinaryTree *tree)
{
if(tree == NULL) return;
else
{
    middleOrderTree(tree -> leftTree);
    printf("%3d",tree -> root);
    middleOrderTree(tree -> rightTree);
}
}

后序遍历:

//首先后序遍历左（右）子树，再后序遍历右（左）子树，最后访问根
void backOrderTree(BinaryTree *tree)
{
if(tree == NULL) return;
else
{
    backOrderTree(tree -> leftTree);
    backOrderTree(tree -> rightTree);
    printf("%3d",tree -> root);
}
}

显示二叉树:

//显示二叉树
void ShowTree(BinaryTree *tree)
{
BinaryTree *stack[100];
BinaryTree *p;
int level[100][2];
int top,n,i;
int width=4;
if(tree != NULL)
{
    printf("\n二叉树的表示法f:\n");
    top=1;
    stack[top] = tree ;
    level[top][0]= width;
    while(top>0)
    {
        p=stack[top];
        n=level[top][0];
        for(i=1;i<=n;i++)
            printf(" ");
        printf("%d",p -> root);
        for(i = n + 1;i < 60; i += 2)
            printf("*");
        printf("\n\t\t");
        top--;
        if(p -> rightTree != NULL)
        {
            top++;
            stack[top] = p -> rightTree;
            level[top][0] = n+width;
            level[top][1] = 2;
        }
        if(p -> leftTree != NULL)
        {
            top++;
            stack[top] = p -> leftTree ;
            level[top][0]= n + width;
            level[top][1]= 1;
        }
     }
  }
}
